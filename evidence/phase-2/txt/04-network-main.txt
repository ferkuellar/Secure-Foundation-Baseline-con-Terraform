data "aws_availability_zones" "available" {
  state = "available"
}

locals {
  module_tags = merge(
    var.common_tags,
    {
      Module = "network"
    }
  )

  azs = slice(data.aws_availability_zones.available.names, 0, 2)

  public_subnets = {
    for idx, cidr in var.public_subnet_cidrs :
    "public-${idx + 1}" => {
      cidr = cidr
      az   = local.azs[idx]
      name = "${var.name_prefix}-subnet-public-${idx + 1}"
    }
  }

  private_subnets = {
    for idx, cidr in var.private_subnet_cidrs :
    "private-${idx + 1}" => {
      cidr = cidr
      az   = local.azs[idx]
      name = "${var.name_prefix}-subnet-private-${idx + 1}"
    }
  }
}

resource "aws_vpc" "main" {
  cidr_block           = var.vpc_cidr
  enable_dns_support   = true
  enable_dns_hostnames = true

  tags = merge(
    local.module_tags,
    {
      Name = "${var.name_prefix}-vpc-main"
    }
  )
}

resource "aws_internet_gateway" "main" {
  vpc_id = aws_vpc.main.id

  tags = merge(
    local.module_tags,
    {
      Name = "${var.name_prefix}-igw-main"
    }
  )
}

resource "aws_subnet" "public" {
  for_each = local.public_subnets

  vpc_id                  = aws_vpc.main.id
  cidr_block              = each.value.cidr
  availability_zone       = each.value.az
  map_public_ip_on_launch = true

  tags = merge(
    local.module_tags,
    {
      Name = each.value.name
      Tier = "public"
    }
  )
}

resource "aws_subnet" "private" {
  for_each = local.private_subnets

  vpc_id                  = aws_vpc.main.id
  cidr_block              = each.value.cidr
  availability_zone       = each.value.az
  map_public_ip_on_launch = false

  tags = merge(
    local.module_tags,
    {
      Name = each.value.name
      Tier = "private"
    }
  )
}

resource "aws_eip" "nat" {
  domain = "vpc"

  tags = merge(
    local.module_tags,
    {
      Name = "${var.name_prefix}-eip-nat-1"
    }
  )
}

resource "aws_nat_gateway" "main" {
  allocation_id = aws_eip.nat.id
  subnet_id     = aws_subnet.public["public-1"].id

  depends_on = [aws_internet_gateway.main]

  tags = merge(
    local.module_tags,
    {
      Name = "${var.name_prefix}-natgw-1"
    }
  )
}
