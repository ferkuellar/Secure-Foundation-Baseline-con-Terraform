locals {
  module_tags = merge(
    var.common_tags,
    {
      Module = "observability"
    }
  )

  cwagent_parameter_name = "AmazonCloudWatch-${var.name_prefix}-linux"

  host_system_log_group = "/aws/ec2/${var.name_prefix}/system"
  host_auth_log_group   = "/aws/ec2/${var.name_prefix}/auth"
  host_boot_log_group   = "/aws/ec2/${var.name_prefix}/boot"

  cwagent_config = jsonencode({
    agent = {
      metrics_collection_interval = 60
      run_as_user                 = "root"
    }
    metrics = {
      namespace = "CWAgent"
      append_dimensions = {
        InstanceId = "${aws:InstanceId}"
      }
      metrics_collected = {
        mem = {
          measurement = [
            "mem_used_percent"
          ]
        }
        disk = {
          measurement = [
            "disk_used_percent"
          ]
          resources = [
            "/"
          ]
        }
      }
    }
    logs = {
      logs_collected = {
        files = {
          collect_list = [
            {
              file_path      = "/var/log/messages"
              log_group_name = local.host_system_log_group
              log_stream_name = "{instance_id}"
              timezone       = "UTC"
            },
            {
              file_path      = "/var/log/secure"
              log_group_name = local.host_auth_log_group
              log_stream_name = "{instance_id}"
              timezone       = "UTC"
            },
            {
              file_path      = "/var/log/cloud-init.log"
              log_group_name = local.host_boot_log_group
              log_stream_name = "{instance_id}"
              timezone       = "UTC"
            },
            {
              file_path      = "/var/log/cloud-init-output.log"
              log_group_name = local.host_boot_log_group
              log_stream_name = "{instance_id}"
              timezone       = "UTC"
            }
          ]
        }
      }
    }
  })
}

resource "aws_iam_role_policy_attachment" "cloudwatch_agent_server" {
  role       = var.admin_instance_role_name
  policy_arn = "arn:aws:iam::aws:policy/CloudWatchAgentServerPolicy"
}

resource "aws_ssm_parameter" "cwagent_config" {
  name  = local.cwagent_parameter_name
  type  = "String"
  value = local.cwagent_config

  tags = merge(
    local.module_tags,
    {
      Name = "${var.name_prefix}-cwagent-config"
    }
  )
}

resource "aws_ssm_association" "install_and_start_cwagent" {
  name = "AWS-RunShellScript"

  targets {
    key    = "InstanceIds"
    values = [var.admin_instance_id]
  }

  parameters = {
    commands = join("\n", [
      "sudo yum install -y amazon-cloudwatch-agent",
      "sudo /opt/aws/amazon-cloudwatch-agent/bin/amazon-cloudwatch-agent-ctl -a fetch-config -m ec2 -s -c ssm:${aws_ssm_parameter.cwagent_config.name}"
    ])
  }

  depends_on = [
    aws_iam_role_policy_attachment.cloudwatch_agent_server,
    aws_ssm_parameter.cwagent_config,
    aws_cloudwatch_log_group.host_system,
    aws_cloudwatch_log_group.host_auth,
    aws_cloudwatch_log_group.host_boot
  ]
}

resource "aws_cloudwatch_log_metric_filter" "flowlog_rejects" {
  name           = "${var.name_prefix}-flowlog-rejects-filter"
  log_group_name = var.flow_logs_log_group_name
  pattern        = "REJECT"

  metric_transformation {
    name      = "${replace(var.name_prefix, "-", "_")}_flowlog_rejects"
    namespace = "Network/Posture"
    value     = "1"
  }
}
